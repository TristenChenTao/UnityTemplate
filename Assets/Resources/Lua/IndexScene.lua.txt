function start()

	print("indexscene lua start...")

	CS.FairyGUI.GRoot.inst:SetContentScaleFactor(1080, 1920, CS.FairyGUI.UIContentScaler.ScreenMatchMode.MatchWidthOrHeight);               
    CS.FairyGUI.UIPackage.AddPackage("UI/Index");     
    view = CS.FairyGUI.UIPackage.CreateObject("Index", "IndexPage")
    view:SetSize(CS.FairyGUI.GRoot.inst.width,CS.FairyGUI.GRoot.inst.height)
    view:AddRelation(CS.FairyGUI.GRoot.inst, CS.FairyGUI.RelationType.Size)
    CS.FairyGUI.GRoot.inst:AddChild(view)

    rankButton = view:GetChild("n11")
    rankButton.onClick:Add(RankOnClick)

end

end

	print("RankOnClick")
	CS.UnityEngine.SceneManagement.SceneManager.LoadScene("Rank")
end
